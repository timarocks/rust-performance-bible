// From naive to optimized in seconds
fn process_data(data: &[u8]) -> Vec {
    // Naive implementation
    // let result = data.iter().map(|&x| x.wrapping_add(1)).collect();
    
    // Optimized implementation
    let mut result = Vec::with_capacity(data.len());
    unsafe {
        std::ptr::copy_nonoverlapping(
            data.as_ptr(),
            result.as_mut_ptr(),
            data.len()
        );
        result.set_len(data.len());
    }
    result.iter_mut().for_each(|x| *x = x.wrapping_add(1));
    result
}
